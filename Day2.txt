class Solution {
    public boolean isPalindrome(int x) {
        String a = ""+x;
        String b = "";
        for(int i=0;i<a.length();i++){
            b = a.charAt(i)+b;
        }
        if(a.equals(b)) return true;
        
        return false;
    }
}
