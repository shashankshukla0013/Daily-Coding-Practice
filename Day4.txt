class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        num = list(set(nums))
        num.sort()
        for i in range(len(num)):
            nums[i]=num[i]
        for i in range(len(nums)):
            if i>=len(num):
                nums.pop()
        return len(nums)
